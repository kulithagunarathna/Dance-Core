-- 1. Create the Lessons Table
CREATE TABLE IF NOT EXISTS public.lessons (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    title TEXT NOT NULL,
    file_url TEXT NOT NULL,
    is_private BOOLEAN DEFAULT false, -- Useful if you want to gate some PDFs behind login
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 2. Enable Row Level Security (RLS)
ALTER TABLE public.lessons ENABLE ROW LEVEL SECURITY;

-- 3. Create Policies

-- Allow anyone (public) to read/view the lessons
CREATE POLICY "Allow public read access" 
ON public.lessons 
FOR SELECT 
USING (true);

-- Allow only authenticated users to insert/update/delete
-- Note: This assumes you are using Supabase Auth. 
-- Since you are using a custom admin_users table for login, 
-- you might prefer to manage these permissions via your Admin dashboard logic.
CREATE POLICY "Allow admin full access" 
ON public.lessons 
FOR ALL 
TO authenticated 
USING (true);

-- 1. If you haven't enabled RLS yet, do it (you likely already have)
ALTER TABLE public.lessons ENABLE ROW LEVEL SECURITY;

-- 2. Allow anyone to VIEW the lessons (for your students)
CREATE POLICY "Enable read access for all users" 
ON public.lessons FOR SELECT 
USING (true);

-- 3. Allow your Admin App to INSERT lessons 
-- Since you aren't using Supabase Auth, we allow 'anon' role to insert
CREATE POLICY "Enable insert for admin" 
ON public.lessons FOR INSERT 
TO anon 
WITH CHECK (true);

-- 4. Allow your Admin App to DELETE lessons
CREATE POLICY "Enable delete for admin" 
ON public.lessons FOR DELETE 
TO anon 
USING (true);


CREATE TABLE public.admin_users (
  user_id uuid NOT NULL DEFAULT uuid_generate_v4(),
  email text NOT NULL UNIQUE,
  password text NOT NULL,
  role text DEFAULT 'admin'::text,
  is_active boolean DEFAULT true,
  CONSTRAINT users_pkey PRIMARY KEY (user_id)
);
